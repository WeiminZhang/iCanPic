;CAN messages


;--------------------------------------------------------------
;	SYNC
;Sent every 100ms
;If delayed, a byte holds the number of usec late the SYNC is
;--------------------------------------------------------------
CAN_ID_SYNC		EQU	0


;--------------------------------------------------------------
;	CRITICAL_ERROR
;DATA=ErrorType,ErrorLocus,ErrorAction
;--------------------------------------------------------------
CAN_ID_CRITICAL_ERROR	EQU	1



;--------------------------------------------------------------
;	Master node messages
;Begin 2048, end 2303
;--------------------------------------------------------------


;--------------------------------------------------------------
;	CAN_ID_SLAVE_ID_REQ
;A slave has requested an id
;Data will hold a double word identifying the node type
;byte 0=GSI_CAPABILITIES
;	bit0=LCD
;	bit1=KEYPAD
;	bit2=SPI
;	bit3=PIC_ADC	
;	bit4=SPI
;	bit5=I2C
;	bit6=INT
;	bit7=DIO
;Byte 1=GSI_CAPABILITIES
;	bit0=RS232/458
;	
;Byte 2=GENERAL_CAPABILITIES
;	bit0-7=num dacs	
;Byte 3=GENERAL_CAPABILITIES
;	bit0-7=num adcs
;Byte 4= DAC_RESOLUTION
;Byte 5= ADC_RESOLUTION
;Byte 6=num digital inputs
;Byte 7=num digital ouputs
;Byte 8=reserved
;--------------------------------------------------------------

CAN_ID_SLAVE_ID_REQ	EQU	2303
CAN_ID_SLAVE_ID_REQ_LEN	EQU	8	;8 data bytes


;-----------------------------------------------------------------
;		CAN_ID_SLAVE_SET_ID
;
;Each slave will send a CAN_ID_SLAVE_ID_REQ on user request
;Request:User holds down switch between ROW_3 and COL_0 (or '1' on keypad)
;After it sees a SYNC from the master it will send a CAN_ID_SLAVE_SET_ID
;then listen for a CAN_ID_SLAVE_SET_ID
;This will be it's base address for receiving messages the master
;specifically wants to send it.
;------------------------------------------------------------------

CAN_ID_SLAVE_SET_ID		EQU	2304	
CAN_ID_SLAVE_SET_ID_LEN		EQU	4	;The slaves Base ID
;-------------------------------------------------------------------
